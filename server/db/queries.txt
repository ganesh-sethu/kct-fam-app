  CREATE TABLE departments (
        department VARCHAR(50), 
        allocated_budget INT,
        budget_used INT,
        PRIMARY KEY (departmentId)
    );

CREATE TABLE users (
         empId VARCHAR(50), 
         name VARCHAR(50),
         email VARCHAR(50),
         department VARCHAR(50),
         password VARCHAR(255),
         designation VARCHAR(50),
         PRIMARY KEY (empId),
         FOREIGN KEY (department) REFERENCES departments(department)
     );

  CREATE TABLE requests (
        request_id VARCHAR(50), 
        approval_status INT,
        form_data JSON,
        budget INT,
        event_start_date datetime,
        event_end_date datetime,
        rejection_status INT,
        rejection_reason VARCHAR(1000),
        request_msg VARCHAR(1000),
        emp_id VARCHAR(50), 
        PRIMARY KEY (request_id),
        FOREIGN KEY (emp_id) REFERENCES users(emp_id) ON DELETE CASCADE
    );

CREATE TABLE events (
        event_id VARCHAR(50), 
        request_id VARCHAR(50),
        event_proof VARCHAR(255),
        budget_used INT,
        event_status TINYINT,
        PRIMARY KEY (event_id),
        FOREIGN KEY (request_id) REFERENCES requests(request_id) ON DELETE CASCADE
    );
